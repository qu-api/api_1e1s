API调用示例 (Python):


import requests
import json # Import json library for better output

# 替换为您的简易API中转站 Key
API_KEY = "your_jeniya_api_key_xxxxxxxxxx"
# 替换为简易API中转站的通用接口地址 (通常与OpenAI兼容)
# 根据文档，地址可能是 https://api.jeniya.top/v1/chat/completions 或其他
API_URL = "https://api.jeniya.top/v1/chat/completions"

headers = {
    "Authorization": f"Bearer {API_KEY}",
    "Content-Type": "application/json"
}

# 请求体结构与OpenAI等官方API高度兼容
data = {
    # 选择您需要的模型，名称请查阅简易API中转站文档
    # 例如: "gpt-4o", "claude-3-5-sonnet-20240620", "deepseek-coder"
    "model": "gpt-4o",
    "messages": [
        {"role": "system", "content": "你是一个有帮助的AI助手。"},
        {"role": "user", "content": "简述使用简易API中转站的好处。"}
    ],
    # 可以添加其他参数，如 temperature, max_tokens, stream等
    "max_tokens": 500,
    "temperature": 0.7,
    "stream": False # 设置为 True 开启流式传输
}

print(f"正在向 {API_URL} 发送请求...")

try:
    # 发送POST请求
    response = requests.post(API_URL, headers=headers, json=data, timeout=60) # 设置超时

    # 检查HTTP状态码
    response.raise_for_status()

    # 解析并打印响应
    response_json = response.json()
    print("请求成功，响应内容:")
    print(json.dumps(response_json, indent=2, ensure_ascii=False)) # Pretty print JSON

except requests.exceptions.Timeout:
    print("请求超时！")
except requests.exceptions.RequestException as e:
    print(f"API请求出错: {e}")
    if response is not None:
         print(f"HTTP状态码: {response.status_code}")
         print(f"响应体: {response.text}")
except json.JSONDecodeError:
    print("响应不是有效的JSON格式。")
    if response is not None:
        print(f"原始响应文本:\n{response.text}")
except Exception as e:
    print(f"发生未知错误: {e}")
